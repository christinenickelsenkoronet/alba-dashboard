4. Query Cookbook — 12 queries útiles

4.1 KPay GMV Q1 2026 (O3KR2)

SELECT ROUND(SUM(komet_payment_amount)) AS kpay_gmv_q1_2026,
       COUNT(DISTINCT komet_company_id) AS active_merchants
FROM PRODUCTION.ANALYTICS.KORONET_PAYMENTS
WHERE payment_status = 'succeeded'
  AND komet_payment_date BETWEEN '2026-01-01' AND '2026-04-01';

4.2 Network Fees Q1 2026 (O3KR1)

SELECT ROUND(SUM(fee_amount)) AS network_fees_q1_2026
FROM PRODUCTION.ANALYTICS.TRANSACTION_FEES
WHERE ks_flag = TRUE
  AND transaction_type = 'K2K'
  AND transaction_date BETWEEN '2026-01-01' AND '2026-04-01';

4.3 Sales channel mix últimos 12m

SELECT * FROM SEMANTIC_VIEW(
  PRODUCTION.ANALYTICS.SALES_SV
  METRICS COUNT(sale_item_id) AS items, SUM(sales) AS total
  DIMENSIONS sale_details.sales_channel
  WHERE sale_details.shipping_date >= DATEADD(month, -12, CURRENT_DATE)
    AND sale_details.shipping_date < CURRENT_DATE
    AND companies.ks_flag = TRUE
) ORDER BY total DESC;

4.4 Activación post-go-live (TTF Sale por cohort)

WITH cohort AS (
  SELECT company_id, contract_start_date AS go_live
  FROM PRODUCTION.ANALYTICS.COMPANIES
  WHERE ks_flag = TRUE AND contract_start_date BETWEEN '2026-01-01' AND CURRENT_DATE
),
first_tx AS (
  SELECT company_id, MIN(shipping_date) AS first_sale_date
  FROM PRODUCTION.ANALYTICS.SALE_DETAILS
  WHERE shipping_date BETWEEN '2010-01-01' AND CURRENT_DATE
  GROUP BY company_id
)
SELECT c.company_id, c.go_live, ft.first_sale_date,
  DATEDIFF(day, c.go_live, ft.first_sale_date) AS days_to_first_sale
FROM cohort c LEFT JOIN first_tx ft USING (company_id)
ORDER BY days_to_first_sale ASC NULLS LAST;

4.5 Top 10 K2K connections por buyer

SELECT k2k_customer_id AS buyer_company,
       COUNT(DISTINCT k2k_vendor_id) AS sellers_connected,
       COUNT(*) AS total_connections
FROM PRODUCTION.ANALYTICS.K2K_CONNECTIONS
WHERE ks_flag = TRUE AND k2k_connection_status = 'Active'
GROUP BY k2k_customer_id
ORDER BY sellers_connected DESC LIMIT 10;

4.6 Customer Health snapshot (un cliente)

-- Reemplaza :company_id con el target
SELECT
  (SELECT COUNT(*) FROM PRODUCTION.ANALYTICS.USER_STATS
   WHERE company_id = :company_id AND user_stat_date >= DATEADD(day, -30, CURRENT_DATE)) AS active_user_days_30d,
  (SELECT ROUND(SUM(sales)) FROM PRODUCTION.ANALYTICS.SALE_DETAILS
   WHERE company_id = :company_id AND shipping_date >= DATEADD(day, -30, CURRENT_DATE)) AS sales_30d,
  (SELECT COUNT(DISTINCT k2k_connection_id) FROM PRODUCTION.ANALYTICS.K2K_CONNECTIONS
   WHERE k2k_customer_id = :company_id AND k2k_connection_status = 'Active') AS k2k_active_connections,
  (SELECT ROUND(SUM(komet_payment_amount)) FROM PRODUCTION.ANALYTICS.KORONET_PAYMENTS
   WHERE komet_company_id = :company_id AND payment_status = 'succeeded'
     AND komet_payment_date >= DATEADD(day, -30, CURRENT_DATE)) AS kpay_30d;

4.7 Productos con duplicados (por network_code)

SELECT product_network_code_id, product_network_code_name,
       COUNT(DISTINCT product_id) AS variant_count,
       COUNT(DISTINCT company_id) AS account_count
FROM PRODUCTION.ANALYTICS.PRODUCTS
WHERE product_network_code_id IS NOT NULL
GROUP BY product_network_code_id, product_network_code_name
HAVING variant_count > 100
ORDER BY variant_count DESC LIMIT 20;

4.8 KPay adoption rate por industry

WITH kpay_ctas AS (
  SELECT DISTINCT komet_company_id AS company_id
  FROM PRODUCTION.ANALYTICS.KORONET_PAYMENTS WHERE payment_status = 'succeeded'
)
SELECT c.company_industry,
       COUNT(*) AS total_companies,
       COUNT(k.company_id) AS kpay_active,
       ROUND(COUNT(k.company_id)*100.0/COUNT(*), 1) AS adoption_pct
FROM PRODUCTION.ANALYTICS.COMPANIES c
LEFT JOIN kpay_ctas k ON c.company_id = k.company_id
WHERE c.ks_flag = TRUE
GROUP BY c.company_industry ORDER BY adoption_pct DESC;

4.9 Prebook → Sale conversion rate

WITH pb AS (
  SELECT company_id, COUNT(DISTINCT prebook_id) AS prebooks_q
  FROM PRODUCTION.ANALYTICS.PREBOOK_DETAILS
  WHERE shipping_date BETWEEN '2026-01-01' AND '2026-04-01'
  GROUP BY company_id
), sl AS (
  SELECT company_id, COUNT(DISTINCT sale_id) AS sales_q
  FROM PRODUCTION.ANALYTICS.SALE_DETAILS
  WHERE shipping_date BETWEEN '2026-01-01' AND '2026-04-01'
  GROUP BY company_id
)
SELECT pb.company_id, pb.prebooks_q, sl.sales_q,
       ROUND(sl.sales_q * 1.0 / NULLIF(pb.prebooks_q, 0), 2) AS conversion_ratio
FROM pb LEFT JOIN sl USING (company_id) ORDER BY pb.prebooks_q DESC LIMIT 20;

4.10 Digital procurement adoption (corregido)

SELECT
  c.company_industry,
  ROUND(SUM(CASE WHEN pd.sales_channel IN ('Web','Procurement','API') THEN pd.total_cost ELSE 0 END)) AS digital_cost_12m,
  ROUND(SUM(pd.total_cost)) AS total_cost_12m,
  ROUND(SUM(CASE WHEN pd.sales_channel IN ('Web','Procurement','API') THEN pd.total_cost ELSE 0 END)*100.0
        / NULLIF(SUM(pd.total_cost),0), 2) AS digital_adoption_pct
FROM PRODUCTION.ANALYTICS.PROCUREMENT_DETAILS pd
JOIN PRODUCTION.ANALYTICS.COMPANIES c ON pd.company_id = c.company_id
WHERE c.ks_flag = TRUE
  AND pd.shipping_date BETWEEN DATEADD(month,-12,CURRENT_DATE) AND CURRENT_DATE
GROUP BY c.company_industry ORDER BY digital_adoption_pct DESC;

4.11 Customers en credit risk (utilization > 80%)

SELECT cd.customer_id, cd.company_id, cd.total_credit,
       ROUND(SUM(p.payment_amount), 2) AS payments_12m
FROM PRODUCTION.ANALYTICS.CREDIT_DETAILS cd
LEFT JOIN PRODUCTION.ANALYTICS.PAYMENTS p ON cd.customer_id = p.customer_id
WHERE cd.ks_flag = TRUE AND cd.total_credit > 0
GROUP BY cd.customer_id, cd.company_id, cd.total_credit
HAVING payments_12m IS NULL OR payments_12m < cd.total_credit * 0.2
ORDER BY cd.total_credit DESC LIMIT 50;

4.12 PMTs Live 2026 vs Snowflake activación

-- Compara cohort SF (PMTs go-live 2026) vs cohort Snowflake (companies con first sale 2026)
WITH sf_cohort AS (
  SELECT company_id FROM PRODUCTION.ANALYTICS.COMPANIES
  WHERE ks_flag = TRUE AND contract_start_date BETWEEN '2026-01-01' AND CURRENT_DATE
)
SELECT 'SF Live 2026 cohort' AS source, COUNT(*) AS companies FROM sf_cohort
UNION ALL
SELECT 'Activated (≥1 sale)', COUNT(DISTINCT s.company_id)
FROM PRODUCTION.ANALYTICS.SALE_DETAILS s
JOIN sf_cohort c USING (company_id)
WHERE s.shipping_date BETWEEN '2010-01-01' AND CURRENT_DATE;

5. Anomalías & data quality gaps

  1. Fechas corruptas en SALE_DETAILS, PREBOOK_DETAILS, INVOICE_DETAILS: hay rows con shipping_date año 0001-0002 y 7566-8000. Cualquier query nueva debe usar BETWEEN '2010-01-01' AND CURRENT_DATE.
  2. system_type "Unknown" en 21% de companies (128 ctas) + 7 con industry vacío + "Floral - Wholesalers" (con s) duplicando "Wholesaler" en 2.
  3. payment_method en payments tiene 600+ valores en texto libre — bloqueador de cualquier breakdown sin normalizer.
  4. sales_channel "Unknown" en 97% de procurement ($1.46B de $1.51B 12m) — distorsiona ratios de digital adoption.
  5. 78% del catálogo W+I sin product_network_code asignado.
  6. EXPECTED_TRANSACTION_FEES tiene company_id format mixto (numérico + 'C-XXXX'), requiere CAST/parsing en joins.
  7. Sin ks_flag directo en: COMPANY_LOCATIONS, PRODUCTS, VENDORS, PROCUREMENT_DETAILS, EXPECTED_FEES, CONSOLIDATED_TAXES (requieren join con COMPANIES).
  8. KORONET_PAYMENTS sin ks_flag pero tiene payment_status — usar 'succeeded' como filtro.
  9. USER_STATS definición exacta del trigger event no documentada. Pendiente confirmar con data team si es login, TX, o ambos.
  10. Tax tracking en solo 22% de companies (73/332) — gap deliberado de adopción del feature.

6. Conexión con OKRs 2026

O3KR1 — Total transactional fees from network (Q1 2026): $541,269 measurable directo (transaction_fees WHERE transaction_type = 'K2K' AND date in Q1 2026). 191 companies activas en el modelo.
O3KR2 — Payments GMV through Koronet Payments (Q1 2026): calculable directo desde KORONET_PAYMENTS. Lifetime $211.5M, 130 merchants activos. Adopción real ~30-40% (excluyendo Unknown/Free Trial).
Adopción post-go-live: medible cruzando COMPANIES.contract_start_date (cohort) × SALE_DETAILS.shipping_date (first sale). KPI-ACTIV-001 ya hace esto vs cache, pero query directa Snowflake da datos más confiables.

7. Recomendaciones & dashboards futuros

  1. Master catalog Koronet (productos): 4 fases ya propuestas en PROD-ANALYSIS-001 v2. Reduce 17.83M→13M product_id en quick wins, ~6,500 SKUs canónicos en fase final.
  2. Master catalog logístico (boxes): mismo enfoque para 67,795 box_type_id → ~1,500 cajas canónicas. Reduce friction de freight.
  3. Payment_method normalizer: ETL diario que mapea los 600+ valores libres a un set canónico (~15 categorías: ACH, Check, Card, Cash, Adjustment, Wire, etc.). Habilita real digital adoption tracking.
  4. Activation Dashboard v2 (Live 2026 cohort): reemplazar la métrica TTF Sale "blended" por matriz industry × system_type × user_type usando user_by_companies. Resuelve definitivamente IRIS-008.
  5. Customer Risk v3 con prebooks signal: agregar prebooks_count_30d_delta como feature (un cliente que reduce 30% sus prebooks en 30d = early warning de churn).
  6. Network Effect Dashboard: visualizar K2K_CONNECTIONS como grafo + métricas de funnel (In Progress → Vendor Approval → Active conversion rate).
  7. OKR Live Dashboard: single page con O3KR1 ($541K Q1) + O3KR2 ($211.5M lifetime) actualizados diariamente desde Snowflake.
  8. System type cleanup: proyecto data quality para resolver los 128 "Unknown" + parsear el campo como array (no comma-delimited string).
